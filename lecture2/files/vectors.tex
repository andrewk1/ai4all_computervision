\begin{frame}{Vectors}
\begin{itemize}
    \item In logistic regression: $\hat{y} = \sigma(w_0 + w_1x_1 + ... + w_Dx_D)$
    $$\textbf{w} = (w_0, ..., w_D) \in \mathbb{R}^D, \textbf{x} = (x_0, ..., x_D) \in \mathbb{R}^D$$
    $$\hat{y} = \sigma(\textbf{w} \cdot \textbf{x})$$
\end{itemize}
\end{frame}
